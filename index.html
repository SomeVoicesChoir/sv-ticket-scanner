    <script>

        // ============================================

        // CONFIGURATION - Update these values

        // ============================================

        const CONFIG = {

            password: 'scanner2025’,           // Change this to your password

            baseId: 'appDQqklQ8X7LWg8Z',      // Your Airtable Base ID

            tableId: 'tblk1wb8ILiezVLit',     // Your Airtable Table ID

            apiKey: 'patVvXPq0jc2O0VSW.08d384ed54e6826f393cbeb5a76c6485c58602f8607c1b7e746fac421a585f7e'       // Your Airtable PAT token

        };

        // ============================================

        

        let html5QrCode;

        let isScanning = false;

        let stats = {

            totalScanned: 0,

            successCount: 0,

            duplicateCount: 0

        };

        

        // Check if already authenticated

        window.onload = function() {

            if (localStorage.getItem('scanner_authenticated') === 'true') {

                showScanner();

            }

            

            // Allow Enter key to submit password

            document.getElementById('passwordInput').addEventListener('keypress', function(e) {

                if (e.key === 'Enter') {

                    authenticate();

                }

            });

        };

        

        function authenticate() {

            const password = document.getElementById('passwordInput').value;

            const errorMsg = document.getElementById('errorMessage');

            

            if (password === CONFIG.password) {

                localStorage.setItem('scanner_authenticated', 'true');

                errorMsg.style.display = 'none';

                showScanner();

            } else {

                errorMsg.style.display = 'block';

                document.getElementById('passwordInput').value = '';

                document.getElementById('passwordInput').focus();

            }

        }

        

        function logout() {

            localStorage.removeItem('scanner_authenticated');

            if (isScanning) {

                stopScanner();

            }

            document.getElementById('loginScreen').style.display = 'block';

            document.getElementById('scannerInterface').style.display = 'none';

            document.getElementById('passwordInput').value = '';

        }

        

        function showScanner() {

            document.getElementById('loginScreen').style.display = 'none';

            document.getElementById('scannerInterface').style.display = 'block';

        }

        

        function updateStats() {

            document.getElementById('totalScanned').textContent = stats.totalScanned;

            document.getElementById('successCount').textContent = stats.successCount;

            document.getElementById('duplicateCount').textContent = stats.duplicateCount;

        }


        async function startScanner() {

            html5QrCode = new Html5Qrcode("reader");

            

            try {

                await html5QrCode.start(

                    { facingMode: "environment" },

                    {

                        fps: 10,

                        qrbox: { width: 250, height: 250 }

                    },

                    onScanSuccess,

                    onScanError

                );

                

                isScanning = true;

                document.getElementById('startBtn').style.display = 'none';

                document.getElementById('stopBtn').style.display = 'block';

            } catch (err) {

                showResult('error', 'Camera Error', `Failed to start camera: ${err}`);

            }

        }


        async function stopScanner() {

            if (html5QrCode && isScanning) {

                await html5QrCode.stop();

                isScanning = false;

                document.getElementById('startBtn').style.display = 'block';

                document.getElementById('stopBtn').style.display = 'none';

            }

        }


        async function onScanSuccess(decodedText) {

            stats.totalScanned++;

            

            // Pause scanning while processing

            if (isScanning) {

                await stopScanner();

            }

            

            await checkInTicket(decodedText);

            

            // Resume after 2 seconds

            setTimeout(() => {

                if (!isScanning) {

                    startScanner();

                }

            }, 2000);

        }


        function onScanError(error) {

            // Ignore scan errors (happens frequently)

        }


        async function checkInTicket(recordId) {

            try {

                // First, fetch the record to check if already checked in

                const getUrl = `https://api.airtable.com/v0/${CONFIG.baseId}/${CONFIG.tableId}/${recordId}`;

                

                const getResponse = await fetch(getUrl, {

                    headers: {

                        'Authorization': `Bearer ${CONFIG.apiKey}`

                    }

                });

                

                if (!getResponse.ok) {

                    throw new Error('Ticket not found');

                }

                

                const record = await getResponse.json();

                const fields = record.fields;

                

                // Check if already checked in

                if (fields['Checked In']) {

                    stats.duplicateCount++;

                    updateStats();

                    showResult('warning', '⚠️ Already Checked In', 

                        `This ticket was already scanned.`,

                        `Attendee: ${fields['Attendee Name'] || 'Unknown'}<br>

                         Check-in Time: ${fields['Check-in Time'] || 'Unknown'}`

                    );

                    return;

                }

                

                // Check in the ticket

                const updateUrl = `https://api.airtable.com/v0/${CONFIG.baseId}/${CONFIG.tableId}/${recordId}`;

                

                const updateResponse = await fetch(updateUrl, {

                    method: 'PATCH',

                    headers: {

                        'Authorization': `Bearer ${CONFIG.apiKey}`,

                        'Content-Type': 'application/json'

                    },

                    body: JSON.stringify({

                        fields: {

                            'Checked In': true,

                            'Check-in Time': new Date().toISOString()

                        }

                    })

                });

                

                if (!updateResponse.ok) {

                    throw new Error('Failed to update ticket');

                }

                

                stats.successCount++;

                updateStats();

                showResult('success', '✅ Check-In Successful!',

                    `Ticket validated and checked in.`,

                    `Attendee: ${fields['Attendee Name'] || 'Unknown'}<br>

                     Event: ${fields['Event'] || 'Unknown'}`

                );

                

            } catch (error) {

                showResult('error', '❌ Check-In Failed', error.message);

            }

        }


        function showResult(type, title, message, details = '') {

            const resultDiv = document.getElementById('result');

            resultDiv.className = `result ${type}`;

            resultDiv.innerHTML = `

                <h2>${title}</h2>

                <p>${message}</p>

                ${details ? `<div class="ticket-info">${details}</div>` : ''}

            `;

            

            // Auto-hide after 3 seconds for success, keep error/warning visible

            if (type === 'success') {

                setTimeout(() => {

                    resultDiv.style.display = 'none';

                }, 3000);

            }

        }
